 78def normalize(df, dtype: type = np.float32) -> None:
 79
 80    """Normalizes a Pandas Data Frame (DF) for using it with TF2_SOM.
 81
 82    Arguments
 83    ---------
 84        df: pd.DataFrame
 85            Pandas Data Frame.
 86        dtype: type
 87            Neural network data type (default: np.float32).
 88    """
 89
 90    result = df.copy()
 91
 92    for i in df.columns:
 93
 94        max_value = df[i].max()
 95        min_value = df[i].min()
 96
 97        result[i] = (df[i] - min_value) / (max_value - min_value)
 98
 99    return result.astype(dtype)

127    def __init__(self,
128                 m: int, n: int, dim: int,
129                 seed: float = None, dtype: type = np.float32,
130                 learning_rate: float = None, radius: float = None, sigma: float = None, epochs: int = 100, decay_function = _asymptotic_decay):
131
132        """Initializes a Self Organizing Maps.
133
134        A rule of thumb to set the size of the grid for a dimensionality
135        reduction task is that it should contain 5 * sqrt(N) neurons
136        where N is the number of samples in the dataset to analyze.
137
138        Arguments:
139        ----------
140            m: int
141                Number of neuron rows.
142            n: int
143                Number of neuron columns.
144            dim: int
145                Dimensionality of the input data.
146            seed: int
147                Seed of the random generators (default: None).
148            dtype: type
149                Neural network data type (default: np.float32).
150            learning_rate: float
151                Starting value of the learning rate (default: 0.3).
152            radius: float
153                Starting value of the neighborhood radius (default: max(m, n) / 2.0).
154            sigma: float
155                Fixed standard deviation coefficient of the neighborhood function (default: 1.0).
156            epochs: int
157                Number of epochs to train for (default: 100).
158            decay_function: function
159                Function that reduces learning_rate and sigma at each iteration
160                (default: 1.0 / (1.0 + 2.0 * epoch / epochs)).
161        """
162
163        ################################################################################################################
164
165        self._m = m
166        self._n = n
167        self._dim = dim
168        self._seed = seed
169        self._dtype = dtype
170        self._decay_function = decay_function
171
172        ################################################################################################################
173
174        self._rebuild_topography()
175
176        ################################################################################################################
177
178        self._learning_rate = 0.3 if learning_rate is None else dtype(learning_rate)
179        self._radius = max(m, n) / 2.0 if radius is None else dtype(radius)
180        self._sigma = 1.0 if sigma is None else dtype(sigma)
181
182        ################################################################################################################
183
184        self._epochs = abs(int(epochs))
185
186        ################################################################################################################
187
188        self._two = tf.constant(2.00e+00, dtype = dtype)
189
190        self._epsilon = tf.constant(1.00e-20, dtype = dtype)
191
192        self._sqrt_two = tf.constant(1.42e+00, dtype = dtype)
193
194        ################################################################################################################
195
196        self._weights = np.empty(shape = (self._m * self._n, self._dim), dtype = self._dtype)
197
198        self._quantization_errors = np.empty(shape = (self._epochs, ), dtype = self._dtype)
199        self._topographic_errors = np.empty(shape = (self._epochs, ), dtype = self._dtype)

203    def _rebuild_topography(self):
204
205        self._topography = tf.constant(np.array(list(self._neuron_locations(self._m, self._n))))

209    @staticmethod
210    def _neuron_locations(m: int, n: int) -> typing.Iterator[typing.List[int]]:
211
212        for i in range(m):
213
214            for j in range(n):
215
216                yield [i, j]

220    @staticmethod
221    def _argsort_n(x: np.ndarray, n: np.ndarray) -> np.ndarray:
222
223        if n > 1:
224            return tf.nn.top_k(tf.negative(x), k = n).indices
225        else:
226            return tf.expand_dims(tf.argmin(x, axis = 1), axis = 1)

230    def _find_bmus(self, weights: np.ndarray, input_vectors: np.ndarray, n: int = 1) -> typing.List[BMUs]:
231
232        ################################################################################################################
233        # COMPUTE DISTANCE SQUARES                                                                                     #
234        ################################################################################################################
235
236        distance_squares = tf.reduce_sum(
237            tf.square(
238                tf.subtract(
239                    tf.expand_dims(input_vectors, axis = 1),
240                    tf.expand_dims(weights, axis = 0)
241                )
242            ),
243            axis = 2
244        )
245
246        ################################################################################################################
247        # COMPUTE INDICES AND LOCATIONS                                                                                #
248        ################################################################################################################
249
250        result = []
251
252        for bmu_indices in tf.transpose(SOM._argsort_n(distance_squares, n)):
253
254            bmu_locations = tf.gather(self._topography, bmu_indices)
255
256            result.append(SOM.BMUs(bmu_indices, bmu_locations))
257
258        ################################################################################################################
259
260        return result

264    def _train(self, weights: np.ndarray, input_vectors: np.ndarray, epoch: int) -> None:
265
266        ################################################################################################################
267        # SHUFFLE INPUT VECTORS                                                                                        #
268        ################################################################################################################
269
270        shuffled_indices = tf.random.shuffle(tf.range(start = 0, limit = tf.shape(input_vectors)[0], dtype = tf.int64))
271
272        input_vectors = tf.gather(input_vectors, shuffled_indices)
273
274        ################################################################################################################
275        # BEST MATCHING UNITS                                                                                          #
276        ################################################################################################################
277
278        bmus = self._find_bmus(weights, input_vectors, n = 2)
279
280        ################################################################################################################
281        # LEARNING OPERATOR                                                                                            #
282        ################################################################################################################
283
284        decay_function = self._decay_function(epoch, self._epochs)
285
286        current_learning_rate = tf.cast(self._learning_rate * decay_function, dtype = self._dtype)
287        current_radius        = tf.cast(self._radius        * decay_function, dtype = self._dtype)
288
289        ################################################################################################################
290
291        bmu_distance_squares = tf.reduce_sum(
292            tf.square(
293                tf.subtract(
294                    tf.expand_dims(self._topography, axis = 0),
295                    tf.expand_dims(bmus[0].locations, axis = 1)
296                )
297            ),
298            axis = 2
299        )
300
301        ################################################################################################################
302
303        neighbourhood_func = tf.exp(tf.divide(
304            tf.negative(tf.cast(bmu_distance_squares, self._dtype)),
305            tf.multiply(self._two, tf.square(tf.multiply(current_radius, self._sigma)))
306        ))
307
308        ################################################################################################################
309
310        learning_rate_op = tf.multiply(neighbourhood_func, current_learning_rate)
311
312        ################################################################################################################
313        # WEIGHT(EPOCH + 1)                                                                                            #
314        ################################################################################################################
315
316        numerator = tf.reduce_sum(
317            tf.multiply(
318                tf.expand_dims(learning_rate_op, axis = -1),
319                tf.expand_dims(input_vectors, axis = +1)
320            ),
321            axis = 0
322        )
323
324        denominator = tf.expand_dims(tf.reduce_sum(learning_rate_op, axis = 0), axis = -1) + self._epsilon
325
326        ################################################################################################################
327
328        weights.assign(tf.divide(numerator, denominator))
329
330        ################################################################################################################
331        # QUANTIZATION ERROR                                                                                           #
332        ################################################################################################################
333
334        self._quantization_errors[epoch] = tf.reduce_mean(
335            tf.norm(
336                tf.subtract(
337                    input_vectors,
338                    tf.gather(weights, bmus[0].indices)
339                ),
340                axis = 1
341            ),
342            axis = 0
343        )
344
345        ################################################################################################################
346        # TOPOGRAPHIC ERROR                                                                                            #
347        ################################################################################################################
348
349        t = tf.greater(
350            tf.norm(
351                tf.subtract(
352                    tf.cast(bmus[1].locations, dtype = self._dtype),
353                    tf.cast(bmus[0].locations, dtype = self._dtype)
354                ),
355                axis = 1
356            ),
357            self._sqrt_two
358        )
359
360        self._topographic_errors[epoch] = tf.divide(
361            tf.reduce_sum(tf.cast(t, dtype = self._dtype)),
362            tf.cast(tf.shape(input_vectors)[0], dtype = self._dtype)
363        )

367    def train(self, input_vectors: np.ndarray, progress_bar: bool = True) -> None:
368
369        """Trains the trained neural network.
370
371        Parameters
372        ----------
373            input_vectors: np.ndarray
374                Training data.
375            progress_bar: bool
376                Specifying whether a progress bar have to be shown (default: True).
377        """
378
379        ################################################################################################################
380        # SET RANDOM SEED                                                                                              #
381        ################################################################################################################
382
383        if self._seed is not None:
384
385            np.random.seed(self._seed)
386
387            tf.random.set_seed(self._seed)
388
389        ################################################################################################################
390        # INITIALIZE WEIGHTS                                                                                           #
391        ################################################################################################################
392
393        weights_np = np.empty(shape = (self._m * self._n, self._dim), dtype = self._dtype)
394
395        l1 = weights_np.shape[0]
396        l2 = input_vectors.shape[0]
397
398        for i in range(l1):
399
400            j = np.random.randint(l2)
401
402            weights_np[i] = input_vectors[j]
403
404        ################################################################################################################
405
406        weights = tf.Variable(weights_np, dtype = self._dtype)
407
408        ################################################################################################################
409        # TRAIN THE SELF ORGANIZING MAP                                                                                #
410        ################################################################################################################
411
412        for epoch in tqdm.tqdm(range(self._epochs), disable = not progress_bar):
413
414            self._train(weights, input_vectors, epoch)
415
416        ################################################################################################################
417
418        self._weights = weights.numpy()

583    def distance_map(self) -> np.ndarray:
584
585        """Returns the distance map of the neural network weights."""
586
587        ################################################################################################################
588
589        centroids = self.get_centroids()
590
591        ################################################################################################################
592
593        result = np.full(shape = (centroids.shape[0], centroids.shape[1], 8), fill_value = np.nan, dtype = self._dtype)
594
595        ii = 2 * [[0, -1, -1, -1, 0, +1, +1, +1]]
596        jj = 2 * [[-1, -1, 0, +1, +1, +1, 0, -1]]
597
598        for x in range(centroids.shape[0]):
599
600            for y in range(centroids.shape[1]):
601
602                w_2 = centroids[x, y]
603
604                e = y % 2 == 0
605
606                for k, (i, j) in enumerate(zip(ii[e], jj[e])):
607
608                    if 0 <= x + i < centroids.shape[0] \
609                            and \
610                            0 <= y + j < centroids.shape[1]:
611
612                        diff_w_2_w_1 = w_2 - centroids[x + i, y + j]
613
614                        result[x, y, k] = np.sqrt(np.dot(diff_w_2_w_1, diff_w_2_w_1.T))
615
616        result = np.nansum(result, axis = 2)
617
618        return result / result.max()

265    def _train(self, weights: np.ndarray, input_vectors: np.ndarray, epoch: int) -> None:
266
267        ################################################################################################################
268        # SHUFFLE INPUT VECTORS                                                                                        #
269        ################################################################################################################
270
271        shuffled_indices = tf.random.shuffle(tf.range(start = 0, limit = tf.shape(input_vectors)[0], dtype = tf.int64))
272
273        input_vectors = tf.gather(input_vectors, shuffled_indices)
274
275        ################################################################################################################
276        # BEST MATCHING UNITS                                                                                          #
277        ################################################################################################################
278
279        bmus = self._find_bmus(weights, input_vectors, n = 2)
280
281        ################################################################################################################
282        # LEARNING OPERATOR                                                                                            #
283        ################################################################################################################
284
285        decay_function = self._decay_function(epoch, self._epochs)
286
287        current_learning_rate = tf.cast(self._learning_rate * decay_function, dtype = self._dtype)
288        current_radius        = tf.cast(self._radius        * decay_function, dtype = self._dtype)
289
290        ################################################################################################################
291
292        bmu_distance_squares = tf.reduce_sum(
293            tf.square(
294                tf.subtract(
295                    tf.expand_dims(self._topography, axis = 0),
296                    tf.expand_dims(bmus[0].locations, axis = 1)
297                )
298            ),
299            axis = 2
300        )
301
302        ################################################################################################################
303
304        neighbourhood_func = tf.exp(tf.divide(
305            tf.negative(tf.cast(bmu_distance_squares, self._dtype)),
306            tf.multiply(self._two, tf.square(tf.multiply(current_radius, self._sigma)))
307        ))
308
309        ################################################################################################################
310
311        learning_rate_op = tf.multiply(neighbourhood_func, current_learning_rate)
312
313        ################################################################################################################
314        # WEIGHT(EPOCH + 1)                                                                                            #
315        ################################################################################################################
316
317        numerator = tf.reduce_sum(
318            tf.multiply(
319                tf.expand_dims(learning_rate_op, axis = -1),
320                tf.expand_dims(input_vectors, axis = +1)
321            ),
322            axis = 0
323        )
324
325        denominator = tf.expand_dims(tf.reduce_sum(learning_rate_op, axis = 0), axis = -1) + self._epsilon
326
327        ################################################################################################################
328
329        weights.assign(tf.divide(numerator, denominator))
330
331        ################################################################################################################
332        # QUANTIZATION ERROR                                                                                           #
333        ################################################################################################################
334
335        self._quantization_errors[epoch] = tf.reduce_mean(
336            tf.norm(
337                tf.subtract(
338                    input_vectors,
339                    tf.gather(weights, bmus[0].indices)
340                ),
341                axis = 1
342            ),
343            axis = 0
344        )
345
346        ################################################################################################################
347        # TOPOGRAPHIC ERROR                                                                                            #
348        ################################################################################################################
349
350        t = tf.greater(
351            tf.norm(
352                tf.subtract(
353                    tf.cast(bmus[1].locations, dtype = self._dtype),
354                    tf.cast(bmus[0].locations, dtype = self._dtype)
355                ),
356                axis = 1
357            ),
358            self._sqrt_two
359        )
360
361        self._topographic_errors[epoch] = tf.divide(
362            tf.reduce_sum(tf.cast(t, dtype = self._dtype)),
363            tf.cast(tf.shape(input_vectors)[0], dtype = self._dtype)
364        )

584    def distance_map(self) -> np.ndarray:
585
586        """Returns the distance map of the neural network weights."""
587
588        ################################################################################################################
589
590        centroids = self.get_centroids()
591
592        ################################################################################################################
593
594        result = np.full(shape = (centroids.shape[0], centroids.shape[1], 8), fill_value = np.nan, dtype = self._dtype)
595
596        ii = 2 * [[0, -1, -1, -1, 0, +1, +1, +1]]
597        jj = 2 * [[-1, -1, 0, +1, +1, +1, 0, -1]]
598
599        for x in range(centroids.shape[0]):
600
601            for y in range(centroids.shape[1]):
602
603                w_2 = centroids[x, y]
604
605                e = y % 2 == 0
606
607                for k, (i, j) in enumerate(zip(ii[e], jj[e])):
608
609                    if 0 <= x + i < centroids.shape[0] \
610                            and \
611                            0 <= y + j < centroids.shape[1]:
612
613                        diff_w_2_w_1 = w_2 - centroids[x + i, y + j]
614
615                        result[x, y, k] = np.sqrt(np.dot(diff_w_2_w_1, diff_w_2_w_1.T))
616
617        result = np.nansum(result, axis = 2)
